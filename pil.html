The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

# Declare an array that will contain our dictionnary
dico = []

def generateDictionnary(images_path):
    zip_path= images_path
    zip_obj= ZipFile(zip_path)
    files= zip_obj.infolist()

    
    for file in files:
        image=zip_obj.open(file)
        img= Image.open(image)
        img = np.array(img)

        # Convert into grayscale
        gray = cv.cvtColor(img, cv.COLOR_BGR2GRAY)
        
        # Resize image and perform ocr
        scale = 0.7
        gray_resized=cv.resize(gray, (int(img.shape[1]* scale), int(img.shape[0]* scale)),interpolation = cv.INTER_AREA)
        tessdata_dir_config = r'--tessdata-dir "readonly/words_alpha.txt"'
        text = ''
        text = pytesseract.image_to_string(gray_resized)
        
        # Initianlize faces array and clear it before detecting new faces
        faces=[]
        faces.clear()
        
        # Detect faces
        faces = face_cascade.detectMultiScale(gray, 1.2, 8)

        # If faces are detected loop throught them and convert them to thumbnails of size 100px*100px 
        faces_thumbnail = []
        if len(faces) > 0:
            for (x,y,w,h) in faces:
                area = (x,y, x+w, y+h)
                face= img.copy()
                face = Image.fromarray(face).crop(area)
                MAX_SIZE = (100, 100)
                face.thumbnail(MAX_SIZE)
                
                # Add each face to faces array
                faces_thumbnail.append(np.array(face))
        # Add filename, faces array and ocr text to dico array
        dico.append({'file':file.filename,'faces':faces_thumbnail, 'text': text})

import re
import math

def search(query, images_path):
    
    # Empty the dictionnary to make it reusable by another image collection
    dico.clear()
    
    # Perform data collection on the specified images in images_path
    generateDictionnary(images_path)

    number_of_thumbnail_per_row = 5
    thumbnail_width=100    
    
    for i in dico:
        # Generate an invisible contact_sheet by default
        contact_sheet = Image.new('RGB',(1,1), color='white')
        
        # Perform a search in each dico element using the query argument pass to our search function 
        x = re.search(query, i['text'])
        
        # I a text match is found
        if x:
            print('Results found in file',i['file'])
            number_of_thumbnails = len(i['faces'])
            
            # If text has a match but no images are present
            if number_of_thumbnails == 0:
                print('But there were no faces in that file!')
            else:
                # Calculate the number of thumbnails rows needed on our contact sheet and regenerate it with a black color.
                rows = math.ceil(number_of_thumbnails/number_of_thumbnail_per_row)
                contact_sheet = Image.new('RGB',(int(thumbnail_width * number_of_thumbnail_per_row), int(thumbnail_width * rows)), color='black')
                x = 0
                y = 0
                
                # If images were found append each image to the contact sheet and display it.
                if len(i['faces']) > 0:
                    for face in (i['faces']):
                        contact_sheet.paste(Image.fromarray(face), (x,y))
                        x += thumbnail_width
                        if x == thumbnail_width * number_of_thumbnail_per_row:
                            x = 0
                            y += thumbnail_width
            display(contact_sheet)
